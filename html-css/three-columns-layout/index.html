<!DOCTYPE html>

    <meta charset="UTF-8" />
    <meta name="viewport" content="width=device-width, initial-scale=1.0" />
    <title>Document</title>
    <style>
      .container {
        display: table;

      }

        background-color: #bfa;

  </head>
  <body>
    <div class="container">
      <div class="left"></div>
      <div class="main"></div>

    </div>
  </body>
</html>
