# 古代年龄称谓

在考公常识中，古代年龄称谓是文学常识板块的经典考点。从婴儿到期颐，每一个称谓背后都蕴含着深厚的文化内涵。

import { onMounted, onBeforeUnmount } from 'vue'

let chartInstance = null

onMounted(async () => {
  const echarts = await import('echarts')
  const dom = document.getElementById('age-chart')
  if (!dom) return
  
  chartInstance = echarts.init(dom)
  
  const ageData = [
    { value: 0, name: '初度/襁褓', desc: '不满周岁' },
    { value: 2.5, name: '孩提', desc: '2-3岁' },

    { value: 8, name: '总角/垂髫', desc: '童年时期' },
    { value: 10, name: '幼学', desc: '10岁' },
    { value: 12, name: '金钗之年', desc: '女12岁' },
    { value: 13, name: '豆蔻/舞勺', desc: '女13-14/男13岁' },
    { value: 15, name: '志学/及笄/束发', desc: '15岁成年礼' },
    { value: 16, name: '破瓜/二八', desc: '女16岁' },
    { value: 20, name: '弱冠/桃李', desc: '男/女20岁' },
    { value: 24, name: '花信', desc: '女24岁' },
    { value: 30, name: '而立', desc: '30岁' },
    { value: 40, name: '不惑/强仕', desc: '40岁' },
    { value: 50, name: '知天命/艾', desc: '50岁' },
    { value: 60, name: '花甲/耳顺', desc: '60岁' },
    { value: 70, name: '古稀/从心', desc: '70岁' },
    { value: 77, name: '喜寿', desc: '77岁(喜字草书)' },
    { value: 80, name: '耄耋/伞寿', desc: '80-90岁' },
    { value: 88, name: '米寿', desc: '88岁(米字拆分)' },
    { value: 90, name: '鲐背/卒寿', desc: '90岁' },
    { value: 99, name: '白寿', desc: '99岁(百减一)' },
    { value: 100, name: '期颐', desc: '100岁' },
    { value: 108, name: '茶寿', desc: '108岁(廿八十八)' }
  ]

  const option = {
    backgroundColor: 'transparent',
    tooltip: {
      trigger: 'axis',
      formatter: function(p) {
        if (!p || p.length === 0) return '';
        const item = p[0];
        return `<div style="font-weight:bold;color:#4caf50">${item.name}</div>
                年龄：${item.value[0]} 岁左右<br/>
                内涵：${item.data.d || ''}`;
      },
      extraCssText: 'z-index: 100'
    },
    grid: { left: '80', right: '80', bottom: '15%', top: '20%' },
    xAxis: {
      type: 'value',
      min: 0,
      max: 110,
      interval: 10,
      axisLabel: { color: '#ccc', formatter: '{value}岁' },
      splitLine: { lineStyle: { color: 'rgba(255,255,255,0.05)' } }
    },
    yAxis: {
      type: 'value',
      min: 0,
      max: 100,
      show: false
    },
    series: [
      {
        type: 'line',
        smooth: true,
        lineStyle: { color: '#4caf50', width: 3, opacity: 0.6 },
        symbol: 'circle',
        symbolSize: 10,
        itemStyle: { color: '#4caf50', borderColor: '#fff', borderWidth: 2 },
        label: {
          show: true,
          position: 'top',
          formatter: (p) => p.name,
          color: '#ffeb3b',
          fontSize: 11,
          fontWeight: 'bold',
          distance: 10
        },
        data: ageData.map((i, idx) => {
          const yValues = [20, 70, 30, 80, 40, 90, 25, 65, 35, 75, 45, 85, 20, 60, 30, 70, 40, 80, 25, 65, 35, 75, 45, 85];
          const y = yValues[idx % yValues.length];
          return {
            value: [i.value, y],
            name: i.name,
            d: i.desc
          };
        })
      },
      {
        type: 'scatter',
        symbolSize: 18,
        itemStyle: {
          color: 'rgba(76, 175, 80, 0.2)',
          shadowBlur: 8,
          shadowColor: '#4caf50'
        },
        data: ageData.map((i, idx) => {
          const yValues = [20, 70, 30, 80, 40, 90, 25, 65, 35, 75, 45, 85, 20, 60, 30, 70, 40, 80, 25, 65, 35, 75, 45, 85];
          return [i.value, yValues[idx % yValues.length]];
        })
      }
    ]
  }

  chartInstance.setOption(option)
  window.addEventListener('resize', () => chartInstance?.resize())
})

onBeforeUnmount(() => {
  chartInstance?.dispose()
})
</script>
